async function login() { const name = document.getElementById("employeeName").value.trim(); const national_id = document.getElementById("employeeNationalId").value.trim(); // ✅ دخول المسؤول مباشرة بدون سيرفر if (name === "admin" && national_id === "0000") { sessionStorage.setItem("role", "admin"); window.location.href = "/admin-dashboard.html"; return; } try { const response = await fetch(API_BASE + "/login", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ name, national_id }) }); const data = await response.json(); if (!data.success) { showMessage("❌ بيانات غير صحيحة", "red"); return; } sessionStorage.setItem("token", data.token); sessionStorage.setItem("role", data.user.role); window.location.href = "/employee-dashboard.html"; } catch (error) { showMessage("❌ خطأ في الاتصال بالخادم", "red"); } }
